    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    %% DATATYPES
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    
    %% An atom, in Prolog, means a single data item. It may be of one of four types:

    %% a STRING ATOM, like 'This is a string' or

    %% a SYMBOL, like love, john, and pizza, in love(john, pizza). 
    %%          Atoms of this type must start with a lower case letter. 
    %%          They can include digits (after the initial lower-case letter) and the 
    %%          underscore character (_).

    %% the EMPTY LIST []. This is a strange one: other lists are not atoms. 
    %%          If you think of an atom as something that is not divisible into parts 
    %%          (the original meaning of the word atom, though subverted by subatomic 
    %%          physics) then [] being an atom is less surprsing, since it is certainly 
    %%          not divisible into parts.

    %% strings of special characters, like <--->, ..., ===>.When using atoms of this type,
    %%          some care is needed to avoid using strings of special characters with a 
    %%          predefined meaning, like the neck symbol :-, the cut symbol !, and various

    %%          arithmetic and comparison operators. 

    %% The available special characters for constructing this class of atom are: 
    %% +, -, *, /, <, >, =, :, ., &, _, and ~.




    
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    %% UNIFICATIONS
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

    %% [a,b,c] unifies with [Head|Tail] resulting in Head=a and Tail=[b,c]
    %% [a] unifies with [H|T] resulting in H=a and T=[]
    %% [a,b,c] unifies with [a|T] resulting in T=[b,c]
    %% [a,b,c] doesn't unify with [b|T]
    %% [] doesn't unify with [H|T]
    %% [] unifies with []. Two empty lists always match
    



    
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    %% DATATYPES OPERATORS
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    
    var(X).     %% an uninstantiated variable

    nonvar(X).  %% not a variable, or is an instantiated variable
    compound(X).%% a compound term - not an unbound variable, and not atomic (see below)
    atom(X).	%% an atom, or bound to an atom (variables, list, ARE NOT); ([], symbols ARE)
    integer(X).	%% an integer, or bound to an integer
    float(X).	%% a floating point (fractional) number such as 3.5 or 1.0, or bound
                %% to one -but,if float(X) is evaluated using is,then you get this float
    number(X).	%% a number, or bound to a number (integer or float)
    atomic(X).	%% ONE number, ONE string, or ONE atom, or bound to one. Not atomic
                %% atomic(1), atomic(symbol), atomic("lol"), atomic([]) --> true
                %% atomic([1, 2, 3]), atomic([1]) --> false




    



    



    



    
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    %% USING LISTS
    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
    
    
    %% MEMBER - FIND AN ELEMENT IN A LIST
        
        member(X,[X|_]). 
        member(X,[_|T])  :-  member(X,T).
        
            %% Usage: member(2, [1,4,3,4,5]). == false
            %% Usage: member(2, [1,2,3,4,5]). == true
    
    %% NOT MEMBER
    
        non_member(X,[Y|T]) :- X \== Y, non_member(X,T).
        non_member(X,[]).

            % --- More efficient to define non_member separately instead of
            % --- as not(member(X,L))
    
    %% LIST LENGTH
    
        myLength([],Z) :- Z is 0.
        myLength([_|Y], Z) :- myLength(Y,N), Z is N+1. 

            %% Efficient but not using an accumulator (which is tail-recursive!
        
        accLen([X|XS], ACC, L) :- ACC_NEW is ACC+1, accLen(XS, ACC_NEW, L).
        accLen([], L, L).

    %% N-TH ELEMENT OF THE LIST

        nthElement(List, Pos, Answer) :- nthElementAcc(List, Pos, Answer).

        nthElementAcc([X|_], 0, X) :- !.    
        nthElementAcc([X|XS], ACC, L) :- ACC_NEW is ACC-1, nthElementAcc(XS, ACC_NEW, L).

    %% LAST ELEMENT OF THE LIST

        lastElement([], []).
        lastElement([X], X).
        lastElement([X|XS], Answer) :- lastElement(XS, Answer), !.

    %% PENULTIMO ELEMENTO

        %% Define a Prolog procedure to obtain from a list the last but one element.
        %% If the number of elements in the list is less than 2, it should fail.

        lastButOne([X,Y], X) :- !.
        lastButOne([X|XS], ACC) :- lastButOne(XS, ACC).

    %% SPLIT AT
        
        splitAt(String, Index, Left, Right) :- split(String,Index, Left, Right).
        
        split(L,0,[],L).
        split([X|XS], N, [X|ZS], YS) :- N > 0, N1 is N-1, split(XS, N1, ZS, YS).

            %% Split list at one index:
            %%Usage: split([1,2,3,4], 2, Left, Right). %% Left = [1,2] e Right = [3,4]

    %% INSERT AT
    
        insertAt(As, 0, X, [X|As]) :- !.
        insertAt([A|As], I, X, [A|Rs]) :- I>0, I1 is I-1,
            insertAt(As, I1, X, Rs).
        
    %% REMOVE AT
    
        removeAt([X|As], 0, As) :- !.
        removeAt([A|As], I, [A|Rs]) :- I>0, I1 is I-1,
            removeAt(As, I1, Rs).   

    %% REMOVE ELEMENT
   
        delete(A,[A|X],X).
        delete(A,[B|X],[B|Z]) :- delete(A,X,Z).

        % --- make a new set with all members except A 

    %% APPEND LISTS OR UNION OF LIST
    
        append([],L,L).
        append([H|T],L2,[H|L3]) :-  append(T,L2,L3).
        
        %% NON RIMUOVE I DUPLICATI!

        %% E' possibile usare questo script anche all' inverso:
        
           % ?- append(ListLeft, ListRight, [a, b, c]).
            
           % ListLeft  = []
           % ListRight = [a, b, c] ;
            
           % ListLeft  = [a]
           % ListRight = [b, c] ;
            
           % ListLeft  = [a, b]
           % ListRight = [c] ;
            
           % ListLeft  = [a, b, c]
           % ListRight = [] ;
    
    %% FIND IN A LIST

        findInList(_, []) :- fail.
        findInList(Y, [Y|_]).
        findInList(Y, [X|Rest]) :- findInList(Y,X) ; findInList(Y, Rest). % ; means OR

        %%Usage: findInList(a, [root,[a,[b,1,2,3],[c,2],[f,[g,x,y]]]]) ---> true

    %% IS_LIST CHECKS WHENEVER
    
        is_list(X) :- var(X), !, fail.

        is_list([]).
        is_list([_|T]) :- is_list(T).  

        %%Usage: is_list([1,2,"asd",[4,[4,3],5,5]]). is true

    %% INTERSECTION BETWEEN SET

        intersect([ ], _, [ ]).
        intersect([X|R], Y, [X|Z]) :- member(X, Y), !, intersect(R, Y, Z).
        intersect([_|R], Y, Z) :- intersect(R, Y, Z).

        %%Usage: intersection([1,3,5,2,4] ,[6,1,2], X).
        %% X = [1, 2].

    %% UNION BETWEEN SET
        
        union(List1, List2, Answer). %% Already defined! Duplicates are removed :)

        union([],X,X).
        union([X|R],Y,Z) :- member(X,Y), !, union(R,Y,Z).
        union([X|R],Y,[X|Z]) :- union(R,Y,Z).

        %%Usage: union([1,3,5,2,4] ,[6,1,2], X).
        %% X = [3, 5, 4, 6, 1, 2].      

    %% DIFFERENCE BEETWEEN SET

        difference([],Y,[]).
        difference([X|R],Y,Z) :- member(X,Y),!,difference(R,Y,Z).
        difference([X|R],Y,[X|Z]) :- difference(R,Y,Z).

    %% EQUALITY BETWEEN SET

        equal(X,X).
        equal(X,Y) :- difference(X,Y,[]), difference(Y,X,[]).
     
    %% PREFIX

        prefix([X|_], [X]).
        prefix([X|Xs], [X|Z]) :- prefix(Xs, Z).

    %% SUFFIX

        suffix(X,X) :- \+ X = [].
        suffix(List1, List2) :- reverse(List1, TsiL1), reverse(List2, TsiL2), prefix(TsiL1, TsiL2).
        
        % l'ultima riga anche: suffix([_|Xs], S) :- suffix(Xs, S).

    %% CHECK IF SUBSTRING
        
        substring([], _) :- fail.
        substring(Rest1, Rest2) :- suffix(Rest1, X), prefix(X, Rest2), !.

        %% or using mySubset:

        substring(Rest1, Rest2) :- (Rest1, Rest2) ; mySubset(Rest1, Rest2).

    %% GENERATE ALL SUBSETS

        mySubset([], []).
        mySubset([E|Tail], [E|NTail]) :- mySubset(Tail, NTail).
        mySubset([_|Tail], NTail)     :- mySubset(Tail, NTail).     

    %% CHECK IF AT LEAST ONE ELEMENT OF LIST SATISFY A CERTAIN PREDICATE
        
        atLeastOneSatisfies(List, Predicate) :- member(X, List), call(Predicate, X).

        %% Usage:
        %% positive(X) :- number(X), X > 0.
        %% atLeastOneSatisfies([1,2,3], positive) TRUE
        %% atLeastOneSatisfies([-1,-2,-3], positive) FALSE

    %% FIND A SUBSET THAT MATCHES A CERTAIN PREDICATE

        findSubsetThatSatisfies(List, Predicate, Sum) :-  mySubset(List, X), call(Predicate, X).

        %%Usage: 
        %% predicate(X) :- myLength(X, N), N is 5.
        %% findSubsetThatSatisfies([1,2,3], len) :-  mySubset(List, X), call(Predicate, X).

    %% REVERSE A LIST

        rev(List,Answer) :- accRev(List, [], Answer).
        
        accRev([X|XS], ACC, L) :- accRev(XS,[X|ACC], L).
        accRev([],L,L).

    %% REVMAP 
    
        %% Define the operation revmap which reverses a list and then performs a map:
    
        revMap(Function, List, Answer) :- accRevMap(Function, List, [], Answer).

        accRevMap(F, [X|XS], ACC, L) :- call(F, X, W), accRev(XS,[W|ACC], L).
        accRevMap(F, [],L,L).

        %%Usage: 
        %% cube(X,R) :- R is X*X*X.
        %% main :- revMap(cube, [2,3,4], X), print(X), nl.

    %% FLATTEN A LIST

        %% Transform a list, possibly holding lists as elements into a 
        %% 'flat' list by replacing each list with its elements (recursively).
        
        %%Usage: my_flatten([a, [b, [c, d], e]], X) == X = [a, b, c, d, e]
        
        %% Hint: Use the predefined predicates is_list/1 and append/3

        my_flatten(X, [X]) :- \+ is_list(X).
        my_flatten([],[]).
        my_flatten([X|XS], ZS) :- my_flatten(X,Y), my_flatten(XS,YS), append(Y,YS,ZS).

    %% PALINDROME    
    
        %% E' molto facile definire le palindrome: basta verificare che l' inverso
        %% sia uguale alla parola normale!

        palindrom(Word) :- rev(Word,Word).
    
    %% COMPRESS 
        
        % Elimina i duplicati CONSECUTIVI di una lista di elementi
        
        compress([],[]).
        compress([X],[X]).
        compress([X,X|XS],ZS) :- compress([X|XS],ZS).
        compress([X,Y|YS],[X|ZS]) :- X \= Y, compress([Y|YS],ZS).
        
        % e.g. compress([1,1,1,2,2,2,1,1], Answer) == Answer = [1,2,1]
        
    %% REMOVE DUPLICATES AND CREATE A SET
    
        remdupl([],[]).
        remdupl([X|Xs],[X|Ys]) :- \+ (member(X,Xs)), remdupl(Xs,Ys). 
        remdupl([X|Xs],Out) :- member(X,Xs), remdupl(Xs,Out). 
        
        % e.g. remdupl([1,1,1,2,2,2,1,1], Answer) == Answer = [1,2]
   
        make_set([],[]).
        make_set(X,Y) :- setof(Z,member(Z,X),Y).
        
    %% SHOW DUPLICATES
    
        onlydup([],[]).
        onlydup([Y|Xs],[Y|Ys]) :- member(Y,Xs), onlydup(Xs,Ys). 
        onlydup([X|Xs],Ys) :- not(member(X,Xs)), onlydup(Xs,Ys).
        
        % e.g. onlydup([1,1,1,2,5,2,1,1,3,4], Answer) == Answer = [1,2]
        
    %% CHECK IF SECOND ELEMENT IS A PARTICULAR ELEMENT
    
        second(X,[_,X|_]).  
            
            %% Usage: second(2, [1,2,3]). == true
    
    %% CHECK IF TWO LIST ARE IDENTICAL EXCEPT FOR THE FIRST TWO ELEMENT SWAPPED
    
        swap12([A,B|C],[B,A|C]).
        
            %% Usage: swap12([1,2,3,4,5],[2,1,3,4,5].

    %% SUM ALL ELEMENTS 

        sumall([N], N) :- number(N).        
        sumall([X|Rest], N) :- number(X), sumall(Rest, Y), N is X + Y.

            %% Usage: sumall([1,2,3,4], N), N = 10

    %% SPLIT ODDS FROM EVENS
    
        /* split(Xs, Os, Es) is true if Os is a list containing the odd positioned
           elements of the list Xs, and Es is a list containing the even         
           positioned elements of Xs.                                            */
        split([], [], []).
        split([X|Xs], [X|Os], Es):-split(Xs, Es, Os).

    %% EXTRACT A RANDOM NUMBER
    
        rnd_select(_, 0, []).
        rnd_select(XS, N, [X|ZS]) :- N > 0,
